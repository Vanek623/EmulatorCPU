MOV1 5 3//SIZE = 3
MOV1 6 4//BASE = 4
MOV1 4 0//ID = 0
MOV4 6 5//M[BASE]  = SIZE
MOV4 4 5//M[ID] = SIZE
ADD2 4 1//ACC = ID+1
MOV5 4 0//ID = ACC
L1:MOV4 4 4//M[ID] = ID
ADD1 4 6 // ACC = ID + BASE
MOV4 0 0//M[ID+BASE] = ID + BASE
ADD2 4 1//ACC = ID+1
MOV5 4 0//ID = ACC
SUB1 5 4 //ACC =  SIZE - ID
JN L2//JMP TO L2 IF ACC < 0
JMP L1
L2:NOP
