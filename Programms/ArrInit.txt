MOV1 5 3//SIZE = 3
MOV1 6 4//BASE = 4
MOV1 2 0//ID = 0
MOV4 6 5//M[BASE]  = SIZE
MOV4 2 5//M[ID] = SIZE
ADD2 2 1//ACC = ID+1
MOV5 2 0//ID = ACC
L1:MOV4 2 2//M[ID] = ID
ADD1 2 6 // ACC = ID + BASE
MOV4 0 0//M[ID+BASE] = ID + BASE
ADD2 2 1//ACC = ID+1
MOV5 2 0//ID = ACC
SUB1 5 2 //ACC =  SIZE - ID
JN L2//JMP TO L2 IF ACC < 0
JMP L1
L2:NOP
