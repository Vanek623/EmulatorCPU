MOV1 5 3//SET SIZE 3
MOV1 6 0//SET BASE 0
MOV5 2 6//SET ID BASE
MOV4 6 5//SET M[BASE] SIZE
ADD2 2 1//SET ACC ID+1
MOV5 2 0//SET ID ACC
L1:MOV4 2 2//SET M[ID] ID
ADD2 2 1//SET ACC ID+1
MOV5 2 0//SET ID ACC
ADD1 5 6//SET ACC SIZE + BASE
SUB1 0 2//SET ACC SIZE+BASE-ID
JN L2//JMP TO L2 IF ACC < 0
JMP L1
L2:NOP
